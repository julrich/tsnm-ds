## Usage

```tsx
import { Visual } from '@backlight-dev/selection-inventory-n5vl9.blank-kzk2liwl"';

<Visual
  height="default"
  box={{
    enabled: true,
    headline: {
      level: 'p',
      styleAs: 'h2',
      align: 'left',
      content: 'Small Size',
      spaceAfter: 'none',
      pageHeader: false,
    },
    text: 'Hic maxime sed eos non. Consequatur ut qui amet accusantium nesciunt.',
    link: {
      enabled: true,
      variant: 'solid-inverted',
      label: 'Lorem Ipsum',
      size: 'medium',
      iconBefore: false,
      iconAfter: false,
      fillAnimation: false,
      iconAnimation: false,
      href: 'https://example.com',
      newTab: false,
    },
    indent: false,
    horizontal: 'left',
    vertical: 'center',
    background: 'dark',
  }}
  media={{
    mode: 'image',
    image: {
      srcMobile:
        'https://pbs.twimg.com/profile_banners/46692667/1622221867/1500x500',
      srcTablet:
        'https://pbs.twimg.com/profile_banners/46692667/1622221867/1500x500',

      indent: 'none',
    },
  }}
/>;
```
